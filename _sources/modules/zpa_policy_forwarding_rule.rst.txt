zpa_policy_forwarding_rule -- Create a Policy Forwarding Rule.
==============================================================

.. contents::
   :local:
   :depth: 1


Synopsis
--------

This module will create, update or delete a specific Policy Forwarding Rule







Parameters
----------

  client_id (type: str)
    


  client_secret (type: str)
    


  customer_id (type: str)
    


  id (type: str)
    


  name (required, type: str)
    


  description (type: str)
    


  action (type: str, default: INTERCEPT)
    


  action_id (type: str)
    


  default_rule (type: bool)
    


  default_rule_name (type: str)
    


  custom_msg (type: str)
    


  bypass_default_rule (type: bool)
    


  operator (type: str)
    


  policy_type (type: str)
    


  priority (type: str)
    


  rule_order (type: str)
    


  conditions (type: list)
    


    id (type: str)
      


    negated (type: bool)
      


    operator (required, type: str)
      


    operands (type: list)
      


      id (type: str)
        


      idp_id (type: str)
        


      name (type: str)
        


      lhs (required, type: str)
        


      rhs (type: str)
        


      rhs_list (type: list)
        


      object_type (required, type: str)
        




  state (type: str, default: present)
